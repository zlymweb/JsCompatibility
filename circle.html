实现一个Circle类

	//需求：
	a.创建一个圆（Circle）的类，并定义该类的一个属性（半径）和两个方法（周长和面积），其中圆的半径可以通过构造函数初始化
	b.创建圆的一个对象，并调用该对象的方法计算圆的周长和面积

	function Circle(r) {
	    this.r = r ;
	}
	Circle.prototype.zhouChang = function() {
	    return (2*this.r*Math.PI).toFixed(4); //圆周长公式 对结果取小数点后4位（因为π一般取3.1415)
	};
	Circle.prototype.mianJi = function() {
	    return (this.r*this.r*Math.PI).toFixed(4); //圆面积公式 对结果取小数点后4位（因为π一般取3.1415)
	};
	 
	var obj = new Circle(3);
	
	console.log(obj.zhouChang());
	console.log(obj.mianJi());
