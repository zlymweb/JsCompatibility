闭包的应用场景有哪些？

1:保存函数执行状态（保存变量不被释放）

var oLis = document.getElementsByTagName('li');
function helloJs(oLis){
    function result(k){ // result函数就是一个闭包,返回的是一个匿名函数
        return function(){
            alert(oLis[k].innerHTML);
        }
    };
    for(var i = 0; i < oLis.length; i++){
        oLis[i].onclick = result(i);  //每循环一次都会产生一个闭包 result(i)
    }
}
helloJs(oLis);
